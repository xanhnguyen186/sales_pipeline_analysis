Cleaning

Each fix below addresses one issue found above. The steps are chained so the transformation reads as a single, reproducible pipeline.

clean <- raw %>%

  # Remove exact-duplicate rows
  distinct() %>%

  # Standardise Region into canonical labels (accents kept to match the
  # dashboard), mapping the stray city to its region
  mutate(
    Region = str_trim(Region),
    Region = case_when(
      str_to_lower(Region) %in% c("miền bắc", "mien bac")               ~ "Miền Bắc",
      str_to_lower(Region) %in% c("miền nam", "mien nam", "ho chi minh") ~ "Miền Nam",
      str_to_lower(Region) %in% c("miền trung", "mien trung")           ~ "Miền Trung",
      str_to_lower(Region) %in% c("tây nguyên", "tay nguyen")           ~ "Tây Nguyên",
      TRUE ~ Region
    )
  ) %>%

  # Standardise Pipeline_Stage — collapse spelling variants
  mutate(
    Pipeline_Stage = str_trim(Pipeline_Stage),
    stage_key = str_to_lower(str_replace_all(Pipeline_Stage, " ", "")),
    Pipeline_Stage = case_when(
      stage_key == "closedwon"  ~ "Closed Won",
      stage_key == "closedlost" ~ "Closed Lost",
      TRUE ~ Pipeline_Stage
    )
  ) %>%
  select(-stage_key) %>%

  # Correct invalid negative discounts
  mutate(`Discount_%` = if_else(`Discount_%` < 0, abs(`Discount_%`), `Discount_%`)) %>%

  # Replace junk numeric values in Industry
  mutate(
    Industry = case_when(
      is.na(Industry)                   ~ NA_character_,
      str_detect(Industry, "^[0-9.]+$") ~ "Other",
      TRUE ~ Industry
    )
  )
